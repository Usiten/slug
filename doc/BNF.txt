<identifier> ::=	([a-z] | [A-Z] | "_") ([a-z] | [A-Z] | [0-9] | "_")+
<integer>	::=		[0-9]+
<factor>	::=		<integer> | <identifier> | "(" <expr> ")"
<factor_op>	::=		"*" | "/"
<term>		::=		<factor> (<factor_op> <factor>)*
<term_op>	::=		"+" | "-"
<expr>		::=		<term> (<term_op> <term>)*	
<assign>	::=		<identifier> "=" <expr>
<linear>	::=		"=" <expr> ";"
<stmts>		::=		<assign> ";"  (<assign> ";")*
<program>	::=		<stmts>
