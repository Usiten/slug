<integer>	::=		[0-9]+
<factor>	::=		<integer> | "(" <expr> ")"
<factor_op>	::=		"*" | "/"
<term>		::=		<factor> (<factor_op> <factor>)*
<term_op>	::=		"+" | "-"
<expr>		::=		<term> (<term_op> <term>)*	
<exprs>		::=		<expr> ";"  (<expr> ";")*	
<program>	::=		<exprs>
